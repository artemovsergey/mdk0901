                this.findNearestLandmark(userLocation);
            },

            }
        );
    }

    findNearestLandmark(userLocation) {
        let nearestPlace = null;
        let minDistance = Infinity;

        landmarksData.forEach(place => {
            const distance = this.calculateDistance(userLocation, place.coordinates);
            if (distance < minDistance) {
                minDistance = distance;
                nearestPlace = place;
            }
        });

        if (nearestPlace) {
            setTimeout(() => {
                alert(`–ë–ª–∏–∂–∞–π—à–∞—è –¥–æ—Å—Ç–æ–ø—Ä–∏–º–µ—á–∞—Ç–µ–ª—å–Ω–æ—Å—Ç—å: ${nearestPlace.name} (${Math.round(minDistance)} –∫–º)`);
            }, 1000);
        }
    }

    calculateDistance(coord1, coord2) {
        const [lat1, lon1] = coord1;
        const [lat2, lon2] = coord2;
        
        const R = 6371; // –†–∞–¥–∏—É—Å –ó–µ–º–ª–∏ –≤ –∫–º
        const dLat = (lat2 - lat1) * Math.PI / 180;
        const dLon = (lon2 - lon1) * Math.PI / 180;
        
        const a = Math.sin(dLat/2) * Math.sin(dLat/2) +
                Math.cos(lat1 * Math.PI / 180) * Math.cos(lat2 * Math.PI / 180) *
                Math.sin(dLon/2) * Math.sin(dLon/2);
        
        const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a));
        return R * c;
    }

            : '0.0';
        document.getElementById('averageRating').textContent = averageRating;
    }
